정렬_Sort

/*정렬 함수 sort (시간복잡도: nlogn)*/
#include <stdio.h>
#include <algorithm>
// 위의 헤더 파일 필요
using namespace std;
// 위의 소스 코드 필요
// 작성하지 않으면 sort, greater 등의 앞에 std::를 붙여줘야 함
int arr[2301];

int main()
{
	int n=0;
	scanf("%d", &n);
	for(int i=1;i<=n;i++) {
		scanf("%d", &arr[i]);
	}
	sort(arr+1, arr+n+1);
	// sort(배열 이름+정렬 시작 인덱스, 배열 이름+정렬 마지막 인덱스+1);
	for(int i=1;i<=n;i++) {
		printf("%d ", arr[i]);
	}
	return 0;
}

시간_Time

/*시간*/
#include <stdio.h>
#include <time.h>
// 위의 헤더 파일 필요
int n;
long long sum;

int main()
{
	scanf("%d", &n);
  time_t start=clock();
  // time_t는 시간의 자료형
  printf("start: %ld\n", start);
  for(int i=1;i<=n;i++){
	  sum +=i;
	}
  printf("%lld\n", sum);
  time_t end=clock();
  printf("end: %ld\n", end);
  printf("%f\n", (float)(end-start)/CLOCKS_PER_SEC);
  return 0;
}

스택_Stack

/*스택*/
#include <stdio.h>
#include <stack>
// 위의 헤더 파일 필요
using namespace std;
stack <int> st;

int main()
{
	int n;
	scanf("%d", &n);
  st.push(n);
  // 스택에 데이터 삽입
  printf("%d\n",st.top());
  // 가장 마지막에 스택에 삽입된 데이터 확인
  st.pop();
  // 가장 마지막에 스택에 삽입된 데이터 삭제
  printf("%d\n",st.top());
  // 가장 마지막에 스택에 삽입된 데이터 확인
  printf("%d\n",st.size());
  // 스택에 있는 원소의 수
  printf("%d\n",st.empty());
  // 스택이 비어있으면 1 아니면 0을 리턴
}

큐_Queue

/*큐*/
#include <stdio.h>
#include <queue>
// 위의 헤더 파일 필요
using namespace std;
queue <int> q;

int main()
{
	  int n;
	  scanf("%d", &n);
    q.push(n);
    // 큐에 데이터 삽입
    printf("%d\n",q.front());
    // 처음 삽입된 데이터를 리턴
    q.pop();
    // 처음 삽입된 데이터를 삭제
    printf("%d\n",q.front());
    // 처음 삽입된 데이터를 리턴
    printf("%d\n",q.back());
    // 마지막으로 삽입된 데이터를 리턴
    printf("%d\n",q.size());
    // 큐에 있는 원소의 수
    printf("%d\n",q.empty());
    // 큐가 비어있으면 1 아니면 0을 리턴
}

/*예문*/
#include <iostream>
#include <queue>
// 원래는 헤더 파일 <utlity>에 존재하나, <iostream>에도 있음
using namespace std;
queue<pair<int, int> > q; // pair<int, int> 타입의 요소를 저장하는 큐를 정의
// Dev C++에선 >>를 띄어써야함(> >)
// 순서대로 first, second의 명칭을 가지고 있음

int main(){
    int x, y;
    scanf("%d %d", &x, &y); // 표준 입력으로 두 정수를 입력받음
    q.push(make_pair(x, y)); // 입력받은 정수를 pair로 만들어 큐에 추가

    // 큐의 가장 앞 요소를 참조하여 first와 second 값을 출력
    printf("%d\n", q.front().first);
    printf("%d\n", q.front().second);

    q.pop(); // 큐에서 가장 앞 요소를 제거합니다.

    // 큐에서 요소를 제거한 후, 다시 큐의 가장 앞 요소를 참조하려고 하면
    // 큐가 비어있을 수 있으므로, 빈 큐를 참조하면 정의되지 않은 동작이 발생
    // 여기서는 큐가 비어있는 상태에서 접근하여 예외가 발생

    printf("%d\n", q.front().first); // 큐의 front()가 비어있으면 정의되지 않은 동작이 발생
    printf("%d\n", q.front().second); // 동일하게 정의되지 않은 동작 발생
}

vector

#include <iostream>
#include <vector>
#include <stdio.h>

using namespace std;

int main() {
    vector<int> vec; // 벡터 정의
    int n; // 입력 받을 요소의 개수
    int value; // 입력 받을 각 정수
    int pos; // 삽입할 위치

    // 사용자로부터 입력 받을 요소의 개수 입력
    printf("요소의 개수를 입력하세요: ");
    scanf("%d", &n);

    // 사용자로부터 요소를 입력받아 벡터에 추가
    printf("%d개의 요소를 입력하세요:\n", n);
    for (int i = 0; i < n; ++i) {
        scanf("%d", &value);
        vec.push_back(value); // 벡터의 끝에 요소 추가
    }

    // 벡터의 모든 요소 출력
    printf("push_back 이후의 벡터 요소:\n");
    for (int i = 0; i < vec.size(); ++i) {
        printf("%d ", vec[i]);
    }
    printf("\n");

    // 사용자로부터 삽입할 위치와 값을 입력 받음
    printf("삽입할 위치와 값을 입력하세요: ");
    scanf("%d %d", &pos, &value);

    if (pos >= 0 && pos <= vec.size()) {
        vec.insert(vec.begin() + pos, value); // 벡터의 특정 위치에 요소 삽입
    } else {
        printf("잘못된 위치입니다.\n");
    }

    // 벡터의 모든 요소 출력
    printf("insert 이후의 벡터 요소:\n");
    for (int i = 0; i < vec.size(); ++i) {
        printf("%d ", vec[i]);
    }
    printf("\n");

    // 사용자로부터 삭제할 위치 입력
    printf("삭제할 위치를 입력하세요: ");
    scanf("%d", &pos);

    if (pos >= 0 && pos < vec.size()) {
        vec.erase(vec.begin() + pos); // 벡터의 특정 위치의 요소 삭제
    } else {
        printf("잘못된 위치입니다.\n");
    }

    // 벡터의 모든 요소 출력
    printf("erase 이후의 벡터 요소:\n");
    for (int i = 0; i < vec.size(); ++i) {
        printf("%d ", vec[i]);
    }
    printf("\n");

    // 벡터의 첫 번째 요소와 마지막 요소 출력
    if (!vec.empty()) {
        printf("첫 번째 요소: %d\n", vec.front());
        printf("마지막 요소: %d\n", vec.back());
    } else {
        printf("벡터가 비어 있습니다.\n");
    }

    // 벡터의 크기 변경 (크기를 8로 변경하고 추가된 요소는 0으로 초기화)
    vec.resize(8, 0);

    // 벡터의 모든 요소 출력
    printf("resize 이후의 벡터 요소:\n");
    for (int i = 0; i < vec.size(); ++i) {
        printf("%d ", vec[i]);
    }
    printf("\n");

    // 벡터의 모든 요소 제거
    vec.clear();

    // 벡터가 비어 있는지 확인
    printf("벡터가 비어 있나요? %s\n", vec.empty() ? "네" : "아니오");

    return 0;
}
