Backtracking

한정 조건을 가진 문제를 푸는 전략…

일반적인 구조

#include <bits/stdc++.h>
// 대부분의 헤더파일을 담고 있는 파일

void backtrack(현재 상태) {
    if (기저 사례) {
        // 해답을 찾은 경우
        return;
    }

    for (가능한 모든 선택지) {
        if (조건을 만족하는가?) {
            // 현재 선택지를 추가
            backtrack(새로운 상태); // 다음 단계로 진행

            // 백트래킹: 현재 선택지를 포기하고 이전 상태로 되돌아감
        }
    }
}

int main() {
		// 메인함수
}

ex) 부분 수열의 합

nnn﻿ 개의 정수로 이루어진 수열이 있을 때, 크기가 양수인 부분수열 중에서 그 수열의 원소를 다 더한 값이 sss﻿가 되는 경우의 수를 구하는 프로그램을 작성하시오. (1≤nnn﻿≤20, ∣s∣|s|∣s∣﻿≤1,000,000)

#include <stdio.h>

int n, s, sum, cnt;
// sum: 현재 부분 집합의 합, cnt: 조건을 만족하는 부분 집합의 개수
int arr[30], e[30];
// arr: 주어진 배열, e: 부분 집합을 나타내는 배열 (선택된 원소를 저장)

void f(int k) {
    // k는 현재 선택할 배열 원소의 인덱스를 나타냄
    if (k == n + 1) {
    // 배열의 모든 원소를 선택했을 때 (즉, 마지막 원소까지 선택을 마친 경우)
        sum = 0;
        // 부분 집합의 합을 초기화
        for (int i = 1; i <= n; i++) {
		        // 선택된 부분 집합의 합을 계산
            sum += e[i];
        }
        if (sum == s) // 부분 집합의 합이 목표 합 s와 같다면
            cnt++; // 조건을 만족하는 부분 집합의 개수를 증가
        return; // 이 경우에 대한 처리가 끝났으므로 함수 종료 (재귀 호출 종료)
    }
    
    // 현재 원소 arr[k]를 부분 집합에 포함시키는 경우
    e[k] = arr[k]; // e 배열의 k번째 자리에 현재 원소를 포함시킴
    f(k + 1); // 다음 원소를 처리하기 위해 재귀 호출

    // 현재 원소 arr[k]를 부분 집합에 포함시키지 않는 경우
    e[k] = 0; // e 배열의 k번째 자리에 0을 넣어, 현재 원소를 포함시키지 않음
    f(k + 1); // 다음 원소를 처리하기 위해 재귀 호출
    
    //이진 트리 구조! 
}

int main() {
    scanf("%d", &n);
    // 배열의 크기 n을 입력받음
    scanf("%d", &s);
    // 목표 합 s를 입력받음
    for (int i = 1; i <= n; i++) {
        scanf("%d", &arr[i]); // 배열의 원소를 입력받아 arr 배열에 저장
    }
    f(1); // 첫 번째 원소부터 부분 집합에 포함시킬지 말지 결정하는 재귀 호출 시작

    if (s == 0)
        printf("%d", cnt - 1);
        // s가 0인 경우 공집합을 제외하기 위해 cnt에서 1을 빼줌
    else
        printf("%d", cnt);

    return 0; // 프로그램 종료
}

ex) 모든 순열

nnn﻿ 이 주어졌을 때, 1부터 nnn﻿ 까지의 수로 이루어진 순열을 사전순으로 출력하는 프로그램을 작성하시오. (1≤nnn﻿≤8)

#include <stdio.h>

// 전역 변수 선언
int n; // 순열의 크기 (즉, 숫자의 개수)
int arr[2301]; // 순열을 저장할 배열
int check[2301]; // 숫자의 사용 여부를 기록할 배열

// 순열 생성 함수
void func(int k) {
    // 기저 사례: 배열의 끝에 도달했을 때
    if (k == n + 1) {
        // 현재까지 생성된 순열을 출력
        for (int i = 1; i <= n; i++) {
            printf("%d ", arr[i]);
        }
        printf("\n"); // 순열 출력 후 줄바꿈
        return;
    }
    
    // 가능한 숫자를 사용하여 순열을 생성
    for (int i = 1; i <= n; i++) {
        // 숫자 i가 사용되지 않은 경우
        if (check[i] == 0) {
            check[i] = 1; // 숫자 i를 사용한다고 표시
            arr[k] = i; // 배열 arr의 k번째 위치에 숫자 i를 저장
            func(k + 1); // 다음 위치에 대해 순열 생성
            check[i] = 0; // 숫자 i의 사용을 취소 (백트래킹)
        }
    }
    // 트리 구조(수형도)!
}

int main() {
    // 사용자로부터 n 입력 받기
    scanf("%d", &n);
    
    // 순열 생성을 위한 함수 호출
    func(1);
    
    return 0; // 프로그램 종료
}
